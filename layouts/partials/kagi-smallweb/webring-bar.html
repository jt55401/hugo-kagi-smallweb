{{- $params := .Site.Params.kagismallweb -}} {{- if and $params $params.enabled $params.feedurl -}} {{/* Fetch smallweb.txt from GitHub */}} {{- $url := "https://raw.githubusercontent.com/kagisearch/smallweb/main/smallweb.txt" -}} {{- $result := try (resources.GetRemote $url) -}} {{- with $result.Err -}} {{- warnf "kagi-smallweb: failed to fetch smallweb.txt: %s" . -}} {{- else -}} {{- with $result.Value -}} {{/* Parse lines, filtering blanks */}} {{- $feedUrl := $params.feedurl -}} {{- $lines := slice -}} {{- range split .Content "\n" -}} {{- $line := trim . " \r" -}} {{- if $line -}}{{- $lines = $lines | append $line -}}{{- end -}} {{- end -}} {{- $total := len $lines -}} {{/* Find index of feedUrl — use scratch for mutable state in range */}} {{- $s := newScratch -}} {{- $s.Set "idx" -1 -}} {{- range $i, $line := $lines -}} {{- if eq $line $feedUrl -}}{{- $s.Set "idx" $i -}}{{- end -}} {{- end -}} {{- $idx := $s.Get "idx" -}} {{- if lt $idx 0 -}} {{/* feedUrl not found in list */}} {{- warnf "kagi-smallweb: feedUrl %q not found in smallweb.txt — is your RSS URL listed exactly?" $feedUrl -}} {{- else if lt $total 2 -}} {{/* Degenerate: 0 or 1 entry, nothing useful to show */}} {{- else -}} {{/* Compute prev/next indices with wraparound */}} {{- $prevIdx := mod (add (sub $idx 1) $total) $total -}} {{- $nextIdx := mod (add $idx 1) $total -}} {{/* Extract homepage URLs from feed URLs */}} {{- $prevFeed := index $lines $prevIdx -}} {{- $nextFeed := index $lines $nextIdx -}} {{- $prevParsed := urls.Parse $prevFeed -}} {{- $nextParsed := urls.Parse $nextFeed -}} {{- $prevHome := "" -}} {{- $nextHome := "" -}} {{- if and $prevParsed.Scheme $prevParsed.Host -}} {{- $prevHome = printf "%s://%s" $prevParsed.Scheme $prevParsed.Host -}} {{- end -}} {{- if and $nextParsed.Scheme $nextParsed.Host -}} {{- $nextHome = printf "%s://%s" $nextParsed.Scheme $nextParsed.Host -}} {{- end -}} {{- if and $prevHome $nextHome -}}
← {{ $prevParsed.Host }} · ⬡ Kagi Small Web · Random · {{ $nextParsed.Host }} →
{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
